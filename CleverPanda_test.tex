\newlength{\IHeight}

\usepackage{tikz}
\newcommand{\quack}{%
  \setlength{\IHeight}{\fontcharht\font`I}%
  \rule{\CleverPanda{}}{\IHeight}
  \CleverPandaSet
  \begin{tikzpicture}[baseline]
    \draw[line width=\CleverPandaGet{}]
      (0,0) -- (\IHeight,\IHeight)
      (0,\IHeight) -- (\IHeight,0);
  \end{tikzpicture}%
}
