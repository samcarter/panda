\usepackage{tikz}
\newlength{\IHeight}
\newcommand{\quack}{%
  \setlength{\IHeight}{\fontcharht\font`I}%
  \rule{\BusyPanda{0.1}\IHeight}{\IHeight}
  \begin{tikzpicture}[baseline]
    \draw[line width=\BusyPanda{0.1}\IHeight]
      (0,0) -- (\IHeight,\IHeight)
      (0,\IHeight) -- (\IHeight,0);
  \end{tikzpicture}%
}
